	\Huge\textbf{Motivation}

\begin{frame}{State Estimation of Dynamic Systems}

\begin{itemize}
    \item Many cyber-physical systems rely on \textbf{estimating hidden dynamical states over time}
    \item Typical examples: 
    \begin{itemize}
        \item Target tracking (position + velocity)
        \item Environmental monitoring (temperature, pollution fields)
    \end{itemize}
    \item In all these cases:
    \begin{itemize}
        \item \textbf{the system state is not directly observable}
        \item \textbf{we only get noisy and partial measurements}
    \end{itemize}
\end{itemize}
\vspace{0.5cm}

Core problem: \alert{\textit{reconstruct latent dynamics from uncertain observations}}
\end{frame}

\begin{frame}{Bayesian Inference}
\begin{itemize}
    \item Goal: \textbf{estimation of unknown parameters ($\theta$) from observed data ($D$)}
    \item \textit{Prior} $P(\theta)$: belief about the parameter before observing data
    \item \textit{Likelihood} $P(D \mid \theta)$: how compatible the observed data is with a given parameter value
    \item \textit{Posterior} $P(\theta \mid D)$: updated belief after observing data
\end{itemize}

\vspace{0.5cm}

\begin{center}
$P(\theta \mid D) = \frac{P(D \mid \theta) P(\theta)}{P(D)}$
\end{center}

\vspace{0.5cm}

\textit{Intuition:} start from a prior belief, observe data, and update the estimate accordingly
\end{frame}

%------------------------------------------------

\begin{frame}{From Bayesian Inference to Particle Filters}

In principle, Bayesian inference provides a complete solution to state estimation:

\begin{center}
$P(x_t \mid y_1, \dots, y_t)$
\end{center}

Where:
\begin{itemize}
  \item $y_1, \dots, y_t$ are the \textbf{noisy observations} up to time $t$
  \item $x_t$ is the \textbf{hidden state} at time $t$
\end{itemize}

\vspace{0.5cm}

However, computing this distribution exactly is \textbf{tractable} only under strong assumptions:

\begin{itemize}
  \item Linear system dynamics
  \item Linear observation model
  \item Gaussian noise
\end{itemize}

\vspace{0.5cm}

Under these conditions, the posterior remains Gaussian and can be computed analytically using the \textit{Kalman filter}.

\vspace{0.6cm}

In many real-world scenarios, these assumptions do not hold:

\begin{itemize}
  \item systems are \textit{nonlinear}
  \item noise is \textit{non-Gaussian}
  \item the posterior can be \textit{complex or multimodal}
\end{itemize}

\vspace{0.6cm}

As a result, exact Bayesian inference becomes intractable, motivating the use of more general filtering methods.
\end{frame}

\begin{frame}[standout]
	\Huge\textbf{Particle Filters}
\end{frame}

\begin{frame}{Centralized Particle Filters}

\begin{itemize}
    \item We assume a centralized setting: all measurements are available to a single estimator
    \item Particle filtering is useful when the posterior is too complex to compute analytically
\end{itemize}

\vspace{0.25cm}

The system evolution and observation model are defined as:

\[
\begin{cases}
x_{k+1} = f_k(x_k, v_k) \\[4pt]
z_k = h_k(x_k, n_k)
\end{cases}
\]

\vspace{0.2cm}

\begin{itemize}
    \item $x_k$: hidden state at time $k$
    \item $z_k$: measurement at time $k$
    \item $f_k$ and $h_k$ may be nonlinear, and noises ($v_k$ and $n_k$) need not be Gaussian
\end{itemize}

\end{frame}

%------------------------------------------------

\begin{frame}{The Bayesian filtering problem}

\begin{itemize}
    \item At time $k$, the objective is to estimate the posterior distribution

        \[
        p(x_k \mid z_{1:k})
        \]

    
    \item Recursive Bayesian filtering has two steps:

        \[
        p(x_k \mid z_{1:k-1})
        =
        \int p(x_k \mid x_{k-1}) p(x_{k-1} \mid z_{1:k-1}) \, dx_{k-1}
        \]

        \[
        p(x_k \mid z_{1:k})
        =
        \frac{
        p(z_k \mid x_k) p(x_k \mid z_{1:k-1})
        }{
        \int p(z_k \mid x_k) p(x_k \mid z_{1:k-1}) \, dx_k
        }
        \]

    \item The first integral is the prediction step
    % \item The denominator in the second equation is the normalization constant.
    \item In nonlinear / non-Gaussian models, these integrals are generally intractable
\end{itemize}
\end{frame}

%------------------------------------------------

\begin{frame}{Why particles?}

\begin{itemize}
    \item Particle filtering replaces intractable integrals with \textbf{Monte Carlo approximations}
    \item The posterior is represented by a finite set of weighted samples
        \[
        p(x_k \mid z_{1:k})
        \approx
        \sum_{i=1}^{N_s} w_k^i \, \delta(x_k - x_k^i)
        \]
    \item Each particle $x_k^i$ is a candidate hypothesis for the state
    \item Each weight $w_k^i$ quantifies how plausible that hypothesis is
    \item The Dirac delta $\delta(\cdot)$ means that probability mass is concentrated on sampled locations
\end{itemize}

\vspace{0.2cm}

\textbf{Particles are not just points: together, they form a discrete approximation of the posterior}

\end{frame}

\begin{frame}{What do particles actually approximate?}

\begin{itemize}
    \item The key point is that particles approximate expectations under the posterior
    \item For any function $g(\cdot)$
        \[
        \int g(x_k) p(x_k \mid z_{1:k}) \, dx_k
        \approx
        \sum_{i=1}^{N_s} w_k^i g(x_k^i)
        \]
    \item Hence, difficult integrals become weighted sums
    \item This is what makes particle methods computationally useful
    \item Important example: posterior mean / MMSE estimate
        \[
        \hat{x}_k
        \approx
        \sum_{i=1}^{N_s} w_k^i x_k^i
        \]
\end{itemize}




\end{frame}

%------------------------------------------------

\begin{frame}{Importance sampling}

\begin{itemize}
    \item We usually cannot sample directly from the posterior $p(x_k \mid z_{1:k})$
    \item Instead, we sample from a proposal distribution $q(x_k \mid z_{1:k})$ and then correct via weights
        \[
        \tilde{x}_k^i \sim q(x_k \mid z_{1:k})
        \]

        \[
        p(x_k \mid z_{1:k})
        \approx
        \sum_{i=1}^{N_s} \tilde{w}_k^i \delta(x_k - \tilde{x}_k^i)
        \]

        \[
        \tilde{w}_k^i
        \propto
        \frac{
        p(\tilde{x}_k^i \mid z_{1:k})
        }{
        q(\tilde{x}_k^i \mid z_{1:k})
        }
        \]
    \item Intuition:
    \begin{itemize}
        \item sample from a tractable distribution
        \item reweight to recover the target posterior
    \end{itemize}
\end{itemize}

\end{frame}


\begin{frame}[allowframebreaks]{High-level Particle Filter Algorithm}

\begin{enumerate}
    \item \textbf{Initialize} a set of particles $\{x_0^i\}_{i=1}^{N_s}$ and associated weights $\{w_0^i\}_{i=1}^{N_s}$
    
    \vspace{0.2cm}
    
    \item For each time step $k = 1, 2, \dots$:
    \begin{enumerate}
        \item \textbf{Prediction:} propagate each particle through the dynamics / proposal model
        \[
        x_k^i \sim q(x_k \mid x_{k-1}^i, z_k)
        \]
        
        \item \textbf{Weighting:} update the importance weights using the new observation
        \[
        w_k^i \propto w_{k-1}^i \,
        \frac{
        p(z_k \mid x_k^i)\, p(x_k^i \mid x_{k-1}^i)
        }{
        q(x_k^i \mid x_{k-1}^i, z_k)
        }
        \]
        
        \item \textbf{Normalize} the weights so that
        \[
        \sum_{i=1}^{N_s} w_k^i = 1
        \]
        
        \item \textbf{Resample}
    \end{enumerate}
    
    \vspace{0.2cm}
    
    \item Compute the state estimate, e.g. the MMSE estimate
    \[
    \hat{x}_k \approx \sum_{i=1}^{N_s} w_k^i x_k^i
    \]
\end{enumerate}

\end{frame}

\begin{frame}[standout]
	\Huge\textbf{Distributed Particle Filters}
\end{frame}

\begin{frame}{Why Distributed Particle Filters?}

\begin{itemize}
    \item In many networked systems, observations are collected by \textbf{multiple spatially distributed agents}
    \item A centralized particle filter assumes that all measurements are available in one place
    \item In practice, this assumption is often \textbf{unrealistic} because of:
    \begin{itemize}
        \item limited bandwidth
        \item energy constraints
        \item latency
        \item topology changes
        \item node and link failures
    \end{itemize}
    \item The challenge is therefore to approximate the global posterior while keeping communication and computation affordable
\end{itemize}

\textbf{Main design question:} how do we distribute filtering without losing too much estimation quality?

\end{frame}

%------------------------------------------------

\begin{frame}{From Centralized PF to Distributed PF}

\begin{itemize}
    \item The filtering logic is conceptually the same:
    \begin{itemize}
        \item prediction
        \item weighting
        \item resampling
    \end{itemize}
    \item The difference is architectural:
    \begin{itemize}
        \item measurements are local
        \item communication is constrained
        \item no node may have full global information
    \end{itemize}
    \item As a consequence, the key issue is no longer only \emph{how to evolve particles},
    but also \emph{where information is processed and how it is exchanged}
\end{itemize}

\vspace{0.3cm}

\textbf{Hence, DPF is mainly a coordination problem on top of the PF machinery}

\end{frame}

%------------------------------------------------

\begin{frame}{A High-Level Taxonomy of DPF Architectures}

\begin{itemize}
    \item A first fundamental distinction is whether a \textbf{fusion center} is present or not
    \item If a fusion center is present:
    \begin{itemize}
        \item local agents process measurements
        \item local results are sent to a central node for fusion
    \end{itemize}
    \item If no fusion center is present, the scheme is decentralized
    \item Decentralized schemes can be further divided into:
    \begin{itemize}
        \item \textbf{Leader-agent based} DPF
        \item \textbf{Consensus-based} DPF
    \end{itemize}
\end{itemize}

\vspace{0.3cm}

\textbf{Main trade-off:} estimation quality vs communication burden vs robustness

\end{frame}

\begin{frame}{Fusion Center-Based DPF}

\begin{itemize}
    \item Each agent processes its own measurements locally
    \item Local posteriors / local estimates are then reported to a \textbf{fusion center}
    \item The fusion center combines the received information and computes the global estimate
\end{itemize}

\vspace{0.3cm}

\textbf{Why this is attractive:}
\begin{itemize}
    \item conceptually simple
    \item easy to organize
    \item global decision is available in one place
\end{itemize}

\vspace{0.3cm}

\textbf{Typical use case:}
\begin{itemize}
    \item applications where global knowledge is only needed at one node
\end{itemize}

\end{frame}

%------------------------------------------------

\begin{frame}{Fusion Center-Based DPF: Pros and Cons}

\textbf{Advantages}
\begin{itemize}
    \item simple architecture
    \item easy fusion logic
    \item often good estimation quality
    \item natural if the application already has a central decision-maker
\end{itemize}

\vspace{0.3cm}

\textbf{Drawbacks}
\begin{itemize}
    \item single point of failure
    \item poor robustness against fusion-center faults
    \item high communication burden near the fusion center
    \item strong dependence on network topology
    \item topology changes may require route reconfiguration
\end{itemize}

\end{frame}

%------------------------------------------------

\begin{frame}{Reducing the Burden: Clustered / Two-Tier Architectures}

\begin{itemize}
    \item A common refinement of the fusion-center architecture is a \textbf{two-tier structure}
    \item Some selected nodes act as \textbf{cluster heads}
    \item Nearby sensors send raw measurements or local summaries to the cluster head
    \item Cluster heads then communicate with the fusion center
\end{itemize}

\vspace{0.3cm}

\textbf{Why useful:}
\begin{itemize}
    \item reduces long-distance communication
    \item avoids requiring every sensor to talk directly to the fusion center
\end{itemize}


\end{frame}

%------------------------------------------------

\begin{frame}{Leader-Agent Based DPF}

\begin{itemize}
    \item No fixed fusion center is assumed
    \item Instead, only a subset of agents is activated for global estimation
    \item A sequence of nodes forms a \textbf{leader-agent path}
    \item Along this path, local information is accumulated and propagated
\end{itemize}

\vspace{0.3cm}

\textbf{Key idea:}
\begin{itemize}
    \item global estimation is not done by everyone
    \item it is delegated to selected nodes, dynamically chosen in the network
\end{itemize}

\end{frame}

%------------------------------------------------

\begin{frame}{Leader-Agent Based DPF: Pros and Cons}

\textbf{Advantages}
\begin{itemize}
    \item avoids a permanently fixed central node
    \item can reduce communication compared with fully decentralized schemes
    \item may exploit informative nodes more efficiently
\end{itemize}

\vspace{0.3cm}

\textbf{Drawbacks}
\begin{itemize}
    \item performance depends on how leaders are selected
    \item dynamic leader selection affects both accuracy and energy consumption
    \item still exposes the system to coordination overhead
    \item less robust than fully decentralized approaches if leadership is fragile
\end{itemize}


\end{frame}

%------------------------------------------------

\begin{frame}{Consensus-Based DPF}

\begin{itemize}
    \item All agents participate in the filtering process
    \item No fusion center and no privileged node are required
    \item Agents exchange information iteratively in order to reach agreement, or at least consistent local beliefs
\end{itemize}

\vspace{0.3cm}

\textbf{High-level intuition:}
\begin{itemize}
    \item each node has only local information
    \item repeated communication spreads this information through the network
    \item local estimates progressively become more globally informed
\end{itemize}

\end{frame}

%------------------------------------------------

\begin{frame}{Consensus-Based DPF: Pros and Cons}

\textbf{Advantages}
\begin{itemize}
    \item high robustness to node failures
    \item enhanced scalability
    \item better adaptability to topology changes
    \item no single point of failure
\end{itemize}

\vspace{0.3cm}

\textbf{Drawbacks}
\begin{itemize}
    \item heavier communication demand
    \item possible delay due to consensus iterations
    \item synchronization and consistency can become nontrivial
    \item in some cases, agreement itself is not the real objective
\end{itemize}


\end{frame}

\begin{frame}{What Makes DPF Hard in Practice?}

\begin{itemize}
    \item Particle filters require \textbf{many particles} to represent the posterior
    \item Exchanging particle-related information across nodes can be very expensive
    \item In networked settings, estimation quality is shaped not only by the model, but also by:
    \begin{itemize}
        \item communication delays
        \item packet drops
        \item topology changes
        \item local computation limits
    \end{itemize}
    \item Therefore, DPF is always a compromise between:
    \begin{itemize}
        \item accuracy
        \item communication overhead
        \item computation complexity
        \item robustness
    \end{itemize}
\end{itemize}

\end{frame}

%------------------------------------------------

\begin{frame}{What Should We Be Careful About?}

\begin{itemize}
    \item \textbf{Communication cost:} exchanging particles, weights, or local likelihoods can dominate the whole method
    \item \textbf{Topology sensitivity:} some architectures break or degrade severely when the network changes
    \item \textbf{Delay and asynchrony:} iterative coordination may become stale before convergence
    \item \textbf{Robustness:} architectures centered around special nodes are more fragile
    \item \textbf{Load imbalance:} some nodes may become communication or computation bottlenecks
    \item \textbf{Information granularity:} raw measurements, local posteriors, particles, and likelihood summaries have very different costs
\end{itemize}

\end{frame}

%------------------------------------------------

\begin{frame}{A Design-Space View of DPF}

\begin{itemize}
    \item DPF is not a single algorithmic recipe
    \item It is \textbf{a family of solutions} defined by a few major design choices:
    \begin{itemize}
        \item where fusion happens
        \item who performs the filtering
        \item what information is exchanged
        \item how often coordination happens
    \end{itemize}
    \item Different choices induce different trade-offs
\end{itemize}

\vspace{0.3cm}

\textbf{In short:}
\begin{itemize}
    \item fusion center $\rightarrow$ simple but fragile
    \item leader-agent $\rightarrow$ lighter but coordination-dependent
    \item consensus-based $\rightarrow$ robust but communication-heavy
\end{itemize}

\end{frame}

\begin{frame}[standout]
	\Huge\textbf{A Field-Based Perspective}

    \vspace{0.8cm}

    \scriptsize
    A. Cortecchia, D. Domini, G. Ciatto, R. Casadei, D. Pianini, M. Viroli, 
    ``Flexible Distributed Particle Filtering for the Internet of Things via Aggregate Computing'', 
    submitted to International Conference On Distributed Computing in Smart Systems and the Internet Of Things (DCOSS-IoT 2026) 

\end{frame}

\begin{frame}{How can a field-based perspective help DPF?}
\begin{itemize}
    \item The main point is \textbf{not} to introduce yet another DPF algorithm
    \item The idea is to expose DPF as a \textbf{configurable coordination problem}
    \item A field-based view helps decouple:
    \begin{itemize}
        \item \textbf{filtering logic} (prediction, weighting, resampling)
        \item from \textbf{coordination choices} (where fusion happens, what is exchanged, how far information propagates)
    \end{itemize}
    \item This makes it easier to adapt the same estimator to:
    \begin{itemize}
        \item different network structures
        \item topology changes and failures
        \item heterogeneous sensing conditions
    \end{itemize}
\end{itemize}

\vspace{0.5em}
\textbf{Take-away:} the field-based view turns architectural assumptions into \emph{design parameters}, rather than hard-coded algorithmic commitments.
\end{frame}

\begin{frame}{What becomes configurable?}
\begin{itemize}
    \item \textbf{Where fusion happens}
    \begin{itemize}
        \item fixed fusion center / elected leader / fully decentralized local filters
    \end{itemize}
    \item \textbf{What information is exchanged}
    \begin{itemize}
        \item particles / weights / posterior summaries / likelihood surrogates / raw measurements
    \end{itemize}
    \item \textbf{How far information propagates}
    \begin{itemize}
        \item only local neighborhood /  selected regions / whole network
    \end{itemize}
    \item \textbf{Who is active in the computation}
    \begin{itemize}
        \item everybody / only relevant regions / leaders / partitions
    \end{itemize}
\end{itemize}
\end{frame}

\begin{frame}{A useful extra degree of freedom: aggregate measurements}
\begin{itemize}
    \item A particularly interesting option is to \textbf{share local measurements}, instead of exchanging particles
    \item Each node still runs its own local particle filter
    \item But the weighting step can exploit an \textbf{aggregated measurement function} built from neighboring observations
\end{itemize}

\vspace{0.4em}
\[
\hat{y}_t = H_{\mathcal{N}(k)}\!\left(\left\{ h_k(x_t, v_{k,t}) \right\}_{k \in \mathcal{N}} \right)
\]

\begin{itemize}
    \item Intuition: nearby sensors collectively behave like a \textbf{distributed sensor}
    \item This is attractive when local sensors are noisy, weak, or only partially informative
    \item Communication can be much cheaper than exchanging particle sets
\end{itemize}
\end{frame}

\begin{frame}{Another benefit: resilient fusion-center behavior without a fixed center}
\begin{itemize}
    \item If a fusion-center architecture is desirable, the fusion role need not be statically assigned
    \item It can be implemented through \textbf{leader election}
    \item This preserves the simplicity of fusion-based estimation while improving:
    \begin{itemize}
        \item fault tolerance
        \item adaptability to topology changes
        \item self-healing after node failures
    \end{itemize}
    \item So, instead of choosing \emph{once and for all} between:
    \begin{itemize}
        \item centralized simplicity
        \item and decentralized robustness
    \end{itemize}
    \item we can move along that spectrum through configuration
\end{itemize}
\end{frame}

\begin{frame}{Field-based view: the big picture}
\begin{itemize}
    \item The contribution is mainly a \textbf{design-space reframing}
    \item DPF is described as a family of field computations where we can compose:
    \begin{itemize}
        \item sensing / aggregation
        \item information dissemination
        \item fusion placement
        \item activation scope
    \end{itemize}
    \item This is especially relevant in IoT-like settings, where assumptions on:
    \begin{itemize}
        \item connectivity,
        \item density,
        \item sensing quality,
        \item and node reliability
    \end{itemize}
    \item are weak and may change over time
\end{itemize}

\vspace{0.5em}
\centering
\textbf{In short:} same estimation goal, but more flexibility in how the distributed system realizes it.
\end{frame}

\begin{frame}{Experimental setup}
\begin{itemize}
    \item Target tracking in a 2D space with a static network of \textbf{25 sensors}
    \item Sensors are deployed on a perturbed spatial grid
    \item The target follows a predefined \textbf{non-linear trajectory}
    \item Sensors observe a signal whose quality degrades with distance
    \item Each sensor executes at \textbf{1 Hz}; experiments last \textbf{3000} simulated seconds
    \item Each configuration is repeated over \textbf{100 random seeds}
\end{itemize}

\vspace{0.5em}
Two experiments:
\begin{enumerate}
    \item \textbf{Neighborhood measurement aggregation}: each node runs a local PF and aggregates neighboring measurements
    \item \textbf{Leader-based fusion center}: one elected leader acts as fusion center, and leader failure is injected during execution
\end{enumerate}
\end{frame}

\begin{frame}{Experiment 1: local PF + aggregated measurements (I)}

\begin{itemize}
    \item Each sensor keeps its \textbf{own local particle filter}
    \item No particle exchange among neighbors
    \item What changes is the \textbf{measurement used for weighting}
    \item Neighboring observations are combined into a richer local view
\end{itemize}

\vspace{0.5em}
Main variable:
\[
|\mathcal{N}| \in \{0,1,4,7\}
\]

\begin{itemize}
    \item With very small neighborhoods, filters struggle to converge
    \item Increasing neighborhood size improves information quality
    \item Better measurements $\Rightarrow$ better trajectory reconstruction
\end{itemize}

\end{frame}

\begin{itemize}
    \item aggregating raw local observations can substantially improve weighting quality
    \item and it does so \textbf{without} exchanging particle sets
\end{itemize}
\end{frame}

\begin{frame}{Experiment 2: elected leader as fusion center (I)}

\begin{itemize}
    \item In the second experiment, an elected leader plays the role of fusion center
    \item Measurements are progressively aggregated toward the leader
    \item A \textbf{leader failure} is injected at time step 1500
\end{itemize}

\vspace{0.5em}
What happens:
\begin{itemize}
    \item an initial transient appears during the first leader election
    \item another transient appears when the current leader fails
    \item after re-election, the new leader resumes tracking
\end{itemize}

\vspace{0.5em}
\textbf{Key message:} fusion-center behavior can be retained, but with self-healing and without a permanently fixed central node

\end{frame}
